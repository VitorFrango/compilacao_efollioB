# Definição de uma função para calcular a média de uma lista de números
def float avg(int[] list):
    int sum = 0
    for i in list:
        sum = sum + i
    if size(list) > 0:
        return sum / size(list)
    else:
        return 0.0

# Programa principal
int[] numbers = [1, 2, 3, 4, 5]
print("A média é ", avg(numbers))

# Cálculo de fatorial usando while
def int factorial(int n):
    int result = 1
    while n > 1:
        result = result * n
        n = n - 1
    return result

int num = 5
print("O fatorial de ", num, " é ", factorial(num))

# Condições e loops adicionais
if num > 4:
    print("Número maior que 4")
elif num == 4:
    print("Número é 4")
else:
    print("Número menor que 4")
